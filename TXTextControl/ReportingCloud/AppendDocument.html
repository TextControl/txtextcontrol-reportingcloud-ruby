def to_camelized_hash
  result = {
    "document" => @document,
    "documentDivider" => "None"
  }

  case @document_divider
  when :new_paragraph
    result["documentDivider"] = "NewParagraph"
  when :new_section
    result["documentDivider"] = "NewSection"
  end

  return result
end
