def self.from_camelized_hash(hash)
  l = hash["length"]
  s = hash["start"]
  txt = hash["text"]
  id = hash["isDuplicate"]
  lan = hash["language"]
  return IncorrectWord.new(l, s, txt, id, lan)
end
